def authenticate(credentials)
  OpenStack::Connection.create ({
    username:   credentials[:username],
    api_key:    credentials[:password],
    auth_url:   credentials[:auth_url],
    authtenant: credentials[:tenant],
    region:     credentials[:region]
  })
end
